# Toán tử
## Toán tử sô học
+ Cộng: +, -, *, **(lũy thừa), /, %, ++, --(prefix, postfix);
## Toán tử gán
+ =, +=, -=, *=, /=, **=;
## Toán tử nối chuỗi
+ "Van" + " " + "Vuong" => Van Vuong.
## Toán tử so sánh.
+ ==, !=, >, >=, <, <=.
## Boolean.(true, false)
+ canBuyAlcolhol = age >= 18.
## If else.
+ false: 0, "", '', Nan, null, undefined.
## Toán tử logic
+ &&, ||, !
## Kiểu dữ liệu
+ Primitive-nguyên thủy: number, string, boolean, undefined, null, symbol.
+ Complex: function, object.
